@startuml

|User|
start
:Enters the Reports page;

|System|
:Gets the employee_id of the User;
:Sends a query to the database to obtain generated reports;

|Database|
:Returns query results;

|System|
:Verifies if any reports have been created before;

if (Is there any report previously generated by the user?) then (yes)
  :Generates each created report box in the page;

  |User|
  if (Does the User desire to download any of the shown reports?) then (yes)
    :Download functionality;
        note left: Go to Send a petition to the Hosting Service for the file;

    stop
  else (no)
    if (Does the User use the report search bar?) then (no)
      stop
    endif
  endif

else (no)
  |System|
  :Displays message "No report has been previously generated";
endif

repeat
  |User|
  :Selects a content_type in the search bar;
  fork
    :Writes a content name or id;
  fork again
    :Selects a start date;
  fork again
    :Selects an end date;
  end fork

  |System|
  if (Is there a content type selected?) then (yes)
    if (Is start_date <= end_date?) then (yes)
      :Waits 300ms (debounce);
      :Combines filters;
      :Sends a query with filters to the database;

      |Database|
      :Returns query results;

      |System|
      :Verifies if any reports match the filters;

      if (Is there any report that matches the filters?) then (yes)
        :Generates each report box of the obtained results;

        |User|
        if (Did the user click the download button?) then (yes)

          repeat
            |System|
            :Sends a petition to the Hosting Service for the file;

            |Hosting Service|
            if (Did the petition go successfully?) then (yes)
              :Returns the file requested;
              |System|
              :Starts the download of the file into the User device;
            else (no)
              |System|
              :Shows error "Error from the hosting service. Try again.";
            endif
          repeat while (Did the petition fail?) is (yes) not (no)

          stop

        else (no)
          
        endif

      else (no)
        |System|
        :Displays message "0 Reports found";
      endif

    else (no)
      |System|
      :Displays error "Invalid date range";
    endif

  else (no)
    |System|
    :Displays message "No content type has been selected";
  endif

repeat while (Does the User want to search again?) is (yes) not (no)

stop
@enduml
